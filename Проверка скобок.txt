#include <iostream>
#include <string>
#include <Windows.h>
using namespace std;

void main()
{
	SetConsoleOutputCP(1251);
	SetConsoleCP(1251);
	setlocale(LC_ALL, "Rus");
	string str = "Hello, my friend";
	getline(cin, str);
	cout << str << endl;
	int a = 0;
	int b = 0;
	int c = 0;
	int A = 0;
	int B = 0;
	int C = 0;
	for (int i = 0; i < str.length(); i++)
	{
		if (str[i] == '(')
		{
			a++;
		}
		if (str[i] == ')')
		{
			A++;
		}
		if (str[i] == '{')
		{
			b++;
		}
		if (str[i] == '}')
		{
			B++;
		}
		if (str[i] == '[')
		{
			c++;
		}
		if (str[i] == ']')
		{
			C++;
		}
	}
	if (a == A & b == B)
	{
		cout << "Программа запустится" << endl;
	}
	else cout << "Программа не запустится" << endl;


	system("pause");
}
